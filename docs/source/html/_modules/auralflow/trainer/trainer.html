[docs]class ModelTrainer(ABC):
    r"""Wrapper class that trains a ``SeparationModel`` model.


    All subclasses must implement the ``full_forward`` and ``scheduler_step``
    methods.

    Args:
        model (SeparationModel): Model to train.
        criterion (Union[nn.Module, Callable[..., Tensor]]): Loss criterion.
        optimizer (Optimizer): Optimizer.
        scheduler (object): LR scheduler. See
            :mod:`torch.optim.lr_scheduler`.
        use_mixed_precision (bool): If ``True``, enables automatic mixed
            precision. Default: ``True``.
        grad_scaler (optional[GradScaler]): Gradient scaler. Default: ``None``.
        max_grad_norm (optional[float]): Maximum gradient norm used for
            gradient clipping. Gradient clipping is disabled if ``None`` is
            passed in. Default: ``100.0``.
        stop_patience (int): Number of epochs to train before stopping
            if the validation loss does not improve. Default: ``5``.
        min_improvement (float): Minimum improvement in the validation loss
            required to reset the stop patience counter. Default: ``0.01``.
        max_lr_reductions (int): Maximum number of learning rate reductions
            if ``self.scheduler`` is an instance of
            :class:`~torch.optim.lr_scheduler.ReduceLROnPlateau`. Default:
            ``5``.
        checkpoint_dir (optional[str]): Directory to save or load object states
            from. No checkpoints will be saved if a path is not specified.
            Default: ``checkpoints``.
        overwrite_checkpoint (bool): If ``True``, overwrites previous
            checkpoints. Default: ``True``.
        best_checkpoint (bool): If ``True``, saves the best model checkpoint
            as ``best.pth``. Default: ``True``.
        silent_checkpoint (bool): If ``True``, suppresses checkpoint logging.
            Default: ``True``.
        resume_training (bool): If ``True``, resumes training from checkpoint
            files. Otherwise, training is restarted. Default: ``True``.


        use_tensorboard (bool): If ``True``, enables tensorboard. Default:
            ``True``.

        write_iter_loss (bool): If ``True``, logs iteration training loss if
            tensorboard is enabled. Default: ``True``.
        write_epoch_loss (bool): If ``True``, logs epoch training and
            validation loss if tensorboard is enabled. Default: ``True``.
        visualize_weights (bool): If ``True``, logs model weights by layer if
            tensorboard is enabled. Default: ``True``.
        visualize_gradients (bool): If ``True``, logs gradients with respect
            to layers if tensorboard is enabled. Default: ``True``.
        visualize_norm (bool): If ``True``, logs the 2-norm of each
            weight/gradient if tensorboard is enabled. Default: ``True``.
        visualize_waveform (bool): If ``True``, logs target source estimates
            as waveform images if tensorboard is enabled. Default: ``True``.
        visualize_spectrogram (bool): If ``True``, logs target source estimates
            as spectrogram images if model is an instance of a
            ``SpectrogramMaskModel`` and tensorboard is enabled. Default:
            ``True``.
        play_estimate (bool): If ``True``, logs listenable target source
            estimates if tensorboard is enabled. Default: ``True``.
        play_residual (bool): If ``True``, logs listenable background track
            estimates if tensorboard is enabled. Default: ``True``.
        image_save_dir (optional[str]):
        image_save_freq (int): Frequency in epochs at which to save images.
            Default: ``5``.
    """

    train_losses: List[float]
    val_losses: List[float]

    _callbacks: CallbackManager
    _trainer_state: Dict

    def __init__(
        self,
        model: SeparationModel,
        criterion: Union[nn.Module, Callable[..., Tensor]],
        optimizer: Optimizer,
        scheduler: object,
        use_mixed_precision: bool = True,
        grad_scaler: Optional[GradScaler] = None,
        max_grad_norm: Optional[float] = 100.0,
        stop_patience: int = 5,
        min_improvement: float = 0.01,
        max_lr_reductions: int = 5,
        checkpoint_dir: Optional[str] = "checkpoints",

        overwrite_checkpoint: bool = True,
        best_checkpoint: bool = True,
        silent_checkpoint: bool = False,
        resume_training: bool = True,
        use_tensorboard: bool = True,
        write_iter_loss: bool = True,
        write_epoch_loss: bool = True,
        visualize_weights: bool = True,
        visualize_gradients: bool = True,
        visualize_norm: bool = True,
        visualize_waveform: bool = True,
        visualize_spectrogram: bool = True,
        play_estimate: bool = True,
        play_residual: bool = True,
        image_save_dir: Optional[str] = None,
        image_save_freq: int = 5
    ):
        self.model = model
        self.criterion = criterion
        self.optimizer = optimizer
        self.scheduler = scheduler
        self._use_mixed_precision = use_mixed_precision
        self._grad_scaler = grad_scaler

        # Set max gradient norm for clipping.
        if not max_grad_norm:
            self._max_grad_norm = float("inf")
        else:
            self._max_grad_norm = max_grad_norm

        self._stop_patience = stop_patience
        self._min_improvement = min_improvement

        # Only use reductions counter for ReduceLROnPlateau scheduler.
        if not isinstance(self.scheduler, ReduceLROnPlateau):
            self._max_lr_reductions = 1
        else:
            self._max_lr_reductions = max_lr_reductions

        Path(os.getcwd(), checkpoint_dir).mkdir(parents=True, exist_ok=True)
        self.checkpoint_dir = str(Path(os.getcwd(), checkpoint_dir).absolute())

        self._silent_checkpoint = silent_checkpoint
        self._overwrite_checkpoint = overwrite_checkpoint
        self._best_checkpoint = best_checkpoint

        if resume_training:
            # Load previous training state.
            with open(Path(checkpoint_dir, "model_trainer.pth")) as file:
                self._trainer_state = pickle.load(file)
            self.train_losses = self._trainer_state["train_losses"]
            self.val_losses = self._trainer_state["val_losses"]
        else:
            self._trainer_state = {
                "global_step": -1,
                "last_epoch": -1,
                "stop_patience_counter": self._stop_patience,
                "lr_reductions_counter": self._max_lr_reductions,
                "train_losses": [],
                "val_losses": [],
                "best_val_loss": float("inf")
            }

        if use_tensorboard:
            writer = SummaryWriter(
                log_dir=str(Path(image_save_dir, "runs").absolute())
            )
            # Define callbacks.
            self._callbacks = _create_callbacks(
                model=model,
                tensorboard_writer=writer,
                save_dir=image_save_dir,
                save_freq=image_save_freq,
                write_iter_loss=write_iter_loss,
                write_epoch_loss=write_epoch_loss,
                visualize_weights=visualize_weights,
                visualize_gradients=visualize_gradients,
                visualize_norm=visualize_norm,
                visualize_waveform=visualize_waveform,
                visualize_spectrogram=visualize_spectrogram,
                play_audio=play_estimate,
                embed_residual=play_residual
            )
        else:
            self._callbacks = CallbackManager()

[docs]    def train(
        self,
        train_loader: DataLoader,
        val_loader: DataLoader,
        max_epochs: int = 100,
    ) -> None:
        """Runs training.

        Args:
            train_loader (DataLoader): Training set dataloader.
            val_loader (DataLoader): Validation set dataloader.
            max_epochs (int): Maximum number of epochs to train for.
                Default: ``100``.

        """
        last_epoch = self._trainer_state["last_epoch"] + 1
        global_step = self._trainer_state["global_step"] + 1
        stop_epoch = last_epoch + max_epochs
        max_iters = len(train_loader)

        for epoch in range(last_epoch, stop_epoch):
            print(f"Epoch {epoch}/{stop_epoch - 1}", flush=True)
            total_train_loss = 0

            # Set model to training mode.
            self.model.train()
            with ProgressBar(
                train_loader, total=max_iters, desc="train"
            ) as pbar:
                for idx, (mixture, target) in enumerate(pbar):
                    with autocast(enabled=self._use_mixed_precision):

                        # Run forward pass, calculate mini-batch loss.
                        batch_loss = self.full_forward(mixture, target)
                        loss_val = batch_loss.item()
                        total_train_loss += loss_val
                        mean_train_loss = total_train_loss / (idx + 1)
                        self.train_losses.append(mean_train_loss)

                        # Display loss via progress bar.
                        pbar.set_postfix(
                            {
                                "loss": f"{loss_val:6.6f}",
                                "mean_loss": f"{mean_train_loss:6.6f}",
                            }
                        )

                    # Run backprop.
                    batch_loss.backward()
                    # Loss-end/post-backward phase callback.
                    self._callbacks.on_loss_end(
                        named_losses={
                            "batch_loss_train": loss_val,
                            "total_loss_train": total_train_loss,
                        },
                        global_step=global_step
                    )
                    # Update model parameters.
                    self._optimizer_step()
                    # Iteration-end callback.
                    self._callbacks.on_iteration_end(
                        named_losses={
                            "batch_loss_train": loss_val,
                            "total_loss_train": total_train_loss,
                        },
                        global_step=global_step
                    )
                    # Update global step count.
                    global_step += 1

            # Run validation loop.
            mean_val_loss = self.validate(val_loader=val_loader)
            self.val_losses.append(mean_val_loss)

            # Epoch-end callback.
            val_mixture, val_target = next(iter(val_loader))
            self._callbacks.on_epoch_end(
                named_losses={
                    "mean_loss_train": mean_train_loss,
                    "mean_loss_valid": mean_val_loss
                },
                mixture_audio=val_mixture,
                target_audio=val_target,
                epoch=epoch
            )

            # Check whether model should be saved or training should stop.
            delta = self._trainer_state["best_val_loss"] - mean_val_loss
            if delta >= self._min_improvement:
                self._trainer_state["best_val_loss"] = mean_val_loss
                # save_all(
                #     global_step=epoch,
                #     save_model=True,
                #     save_optimizer=True,
                #     save_scheduler=True,
                #     save_grad_scaler=True
                # )
                self._trainer_state["sp_counter"] = self._stop_patience
            else:
                self._trainer_state["sp_counter"] -= 1
                if not self._trainer_state["sp_counter"]:
                    self._trainer_state["lr_reductions_counter"] -= 1
                    self._trainer_state["sp_counter"] = self._stop_patience

            if not self._trainer_state["lr_reductions_counter"]:
                print("No improvement. Stopping training early...")
                break

            # Save trainer state.
            self._trainer_state["last_epoch"] = epoch
            self._trainer_state["global_step"] = global_step
            with open(Path(self.checkpoint_dir, "model_trainer.pth")) as file:
                pickle.dump(self._trainer_state, file)


    def _optimizer_step(self) -> None:
        """Optimizes and updates the model's parameters."""
        self.model.train()
        # Clip gradient.
        _ = nn.utils.clip_grad_norm_(
            self.model.model.parameters(), max_norm=self._max_grad_norm
        )

        # TODO: Gradient scaling.

        self.optimizer.step()
        # Quicker gradient zeroing.
        for param in self.model.model.parameters():
            param.grad = None

[docs]    def validate(self, val_loader: DataLoader) -> float:
        """Given a validation set, runs validation and returns the mean loss.

        Args:
            val_loader (DataLoader): Validation dataloader.

        Returns:
            float: Mean validation loss.
        """
        max_iters = len(val_loader)
        total_loss = mean_loss = 0

        # Set model to validation mode.
        self.model.eval()
        with ProgressBar(
            val_loader, total=max_iters, desc="valid"
        ) as pbar:
            for idx, (mixture, target) in enumerate(pbar):
                with autocast(enabled=self._use_mixed_precision):
                    with torch.no_grad():

                        # Run forward pass, calculate mini-batch loss.
                        batch_loss = self.full_forward(mixture, target)
                        loss_val = batch_loss.item()
                        total_loss += loss_val
                        mean_loss = total_loss / (idx + 1)

                # Display loss.
                pbar.set_postfix(
                    {
                        "loss": f"{loss_val:6.6f}",
                        "mean_loss": f"{mean_loss:6.6f}",
                    }
                )
        return mean_loss



class _DefaultModelTrainer(ModelTrainer):
    """Default ``ModelTrainer`` class."""

    scheduler: ReduceLROnPlateau

    def __init__(self, *args, **kwargs) -> None:
        super(_DefaultModelTrainer, self).__init__(*args, **kwargs)

    def full_forward(self, mixture: Tensor, target: Tensor) -> Tensor:
        if hasattr(self.criterion, "_forward_wrapper"):
            loss = self.criterion._forward_wrapper(
                    model=self.model, mix_audio=mixture, target_audio=target
            )
            return loss
        else:
            raise AttributeError(
                "Criterion must define the _forward_wrapper() method."
            )

    def scheduler_step(self, val_loss: float) -> None:
        """Reduces lr if val loss does not improve enough within a window."""
        self.scheduler.step(val_loss)
