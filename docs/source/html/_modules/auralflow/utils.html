[docs]def copy_config_template(save_dir: Optional[str] = None) -> None:
    """Clones default ``audio_model_template.json`` file to a save directory.

    Args:
        save_dir (Optional[str]): Path to copy ``audio_model_template.json`` to. If ``None``
            is passed in, defaults to current working directory.
            Default: ``None``.
    """
    if save_dir is None:
        save_dir = os.getcwd()
    Path(save_dir).mkdir(exist_ok=True)
    save_filepath = save_dir + "/audio_model_template.json"
    shutil.copy(src=str(config_template_path), dst=save_filepath)




[docs]def load_config(
    filename: str,
    load_model_params: bool = True,
    load_training_params: bool = True,
    load_dataset_params: bool = True,
    load_viz_params: bool = True
) -> dict:
    r"""Loads the contents of a configuration file into a dictionary.

    Args:
        filename (str): Path to the configuration file.
        load_model_params (bool): If ``True``, model params will be loaded
            into the dictionary. Default: ``True``.
        load_training_params (bool): If ``True``, training params will be
            loaded into the dictionary. Default: ``True``.
        load_dataset_params (bool): If ``True``, dataset params will be loaded
            into the dictionary. Default: ``True``.
        load_viz_params (bool): If ``True``, visualizer params will be
            loaded into the dictionary. Default: ``True``.
    Returns:
        dict: Configuration data.

    Raises:
        FileNotFoundError: Raised if the configuration file does not exist.
        IOError: Raised if the configuration file cannot be read.
        KeyError: Raised if a key corresponding to a set of parameters cannot
        be found.
    """
    try:
        with open(filename) as config_file:
            config_data = json.load(config_file)
            if load_model_params and "model_params" not in config_data:
                raise KeyError("Missing 'model_params'.")
            elif load_training_params and "training_params" not in config_data:
                raise KeyError("Missing 'training_params.")
            elif load_dataset_params and "dataset_params" not in config_data:
                raise KeyError("Missing 'dataset_params.")
            elif load_viz_params and "visualizer_params" not in config_data:
                raise KeyError("Missing 'visualizer_params'.")
            return config_data
    except (FileNotFoundError, IOError) as error:
        raise error




[docs]def save_config(config: dict, save_filepath: str) -> None:
    """Saves configuration data to the given filepath.

    Args:
        config (dict): Configuration data.
        save_filepath (str): Path to save.

    Raises:
        IOError: Raised if the configuration file cannot be saved.
    """
    try:
        with open(save_filepath, "w") as config_file:
            return json.dump(config, config_file, indent=4)
    except IOError as error:
        raise error




[docs]def save_object(
    model: SeparationModel,
    obj_name: str,
    global_step: int,
    checkpoint_dir: Optional[str] = None,
    silent: bool = True
) -> None:
    r"""Saves an object's state as a ``.pth`` file under a checkpoint folder.

    Saving is quietly ignored if the model instance does not have
    ``obj_name`` as an attribute.

    Args:
        model (SeparationModel): Separation model.
        obj_name (str): Object to save: ``'model'`` | ``'optimizer'`` |
            ``'scheduler'`` | ``'grad_scaler'``.
        global_step (int): Global step (e.g. epoch number).
        checkpoint_dir (Optional[str]): Path to a checkpoint directory. If
            no path is specified, file is saved in the current directory.
            Default: ``None``.
        silent (bool): If ``True``, checkpoint dialogue is suppressed.
            Default: ``True``.

    Raises:
        OSError: Raised if object cannot be saved.

    Example:

        >>> import auralflow.utils
        >>> my_model = ...
        >>> auralflow.utils.save_object(
        ...     model=my_model,
        ...     obj_name="model",
        ...     global_step=10,
        ...     checkpoint_dir="path/to/checkpoint_dir"
        ... )
    """
    if hasattr(model, obj_name):
        if checkpoint_dir is None:
            checkpoint_dir = os.getcwd()
        # Create the object-specific filename.
        filename = str(
            Path(checkpoint_dir, f"{obj_name}_{global_step}.pth").absolute()
        )
        if not silent:
            print(f"Saving {obj_name}...")

        # Retrieve object's state.
        if obj_name == "model":
            state_dict = getattr(model, obj_name).cpu().state_dict()
            # Transfer model back to GPU if applicable.
            model.model.to(model.device)
        else:
            state_dict = getattr(model, obj_name).state_dict()
        try:
            # Save object's state to filename.
            torch.save(state_dict, f=filename)
            if not silent:
                print(f"  Successful.")
        except OSError as error:
            print(f"  Failed.")
            raise error




[docs]def load_objects(
    named_objects: Dict[str, object],
    checkpoint_dir: str,
    epoch: int,
    device: str = 'cpu'
) -> None:
    """Loads the state of one or more objects corresponding to a given epoch.

    Args:
        named_objects (Dict[str, object]): Labeled objects in the form of a
            dictionary, with the possible keys: ``'model'`` | ``'optimizer'`` |
            ``'scheduler'`` | ``'grad_scaler'``.
        checkpoint_dir (str): Path to a checkpoint directory.
        epoch (int): Epoch number.
        device (str): Device to load object onto. Default: ``'cpu'``.

    Raises:
        OSError: Raised if an object cannot be loaded.
    """
    for name, object in named_objects.items():
        # Get object-specific filename.
        filename = str(
            Path(checkpoint_dir, f"{name}_{epoch}.pth").absolute()
        )
        try:
            # Try to read object state from the given file.
            state_dict = torch.load(filename, map_location=device)
        except (OSError, FileNotFoundError) as error:
            print("  Failed.")
            raise error

        if hasattr(model, obj_name):
            # Load state into object.
            getattr(model, obj_name).load_state_dict(state_dict)
            print(f"  Successful.")
